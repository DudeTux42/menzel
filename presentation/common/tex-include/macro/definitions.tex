% Prompt
% definitions for standard prompts; the working directory is set as a parameter 
% e.g.
% \uprompt{~} -> user@linux:~>
% the free prompt needs 3 parameters: username, hostname, working directory, e.g.
% \newpromp{anke}{host}{/tmp} -> anke@host:/tmp> 
% the name prompt can be used to set a different username and expects 2 parameters
% \myprompt{anke}{/usr/src/linux} -> anke@linux:/usr/src/linux> 
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% User-Prompt
\newcommand{\uprompt}[1]{user@linux:#1$>$ }
% Root-Prompt
\newcommand{\rprompt}[1]{linux:#1 \# }
% Win-Root-Prompt
\newcommand{\winrprompt}[1]{C:\textbackslash#1> }
% Win-Prompt
\newcommand{\winprompt}[2]{#1:\textbackslash#2> }
% Free-Prompt
\newcommand{\newprompt}[3]{#1@#2:#3$>${} }
% Name-Prompt
\newcommand{\myprompt}[2]{#1@linux:#2$>$ }
